/* ============================================================ Cloudflare Worker - Proxy IQAir pour Hanoi Dashboard À coller sur https://workers.cloudflare.com/ ============================================================ */ /* Cache en mémoire (persiste tant que le Worker reste "chaud") */ let cache = { data: null, timestamp: 0 }; const CACHE_TTL = 600 * 1000; /* 10 minutes en ms */ /* Coordonnées Hanoi */ const LAT = '21.0285'; const LON = '105.8542'; export default { async fetch(request, env) { /* env.IQAIR_KEY sera ta clé API stockée comme Secret dans Cloudflare */ const API_KEY = env.IQAIR_KEY; const corsHeaders = { 'Access-Control-Allow-Origin': '*', 'Content-Type': 'application/json; charset=utf-8', 'Cache-Control': 'public, max-age=300' }; /* Sert le cache si frais */ const now = Date.now(); if (cache.data && (now - cache.timestamp < CACHE_TTL)) { return new Response(cache.data, { headers: corsHeaders }); } /* Appel IQAir */ const url = `https://api.airvisual.com/v2/nearest_city?lat=${LAT}&lon=${LON}&key=${API_KEY}`; try { const res = await fetch(url, { headers: { 'User-Agent': 'Mozilla/5.0 Hanoi-Dashboard' } }); const text = await res.text(); if (res.ok) { cache.data = text; cache.timestamp = now; return new Response(text, { headers: corsHeaders }); } else { /* Si erreur API mais cache existant (même périmé), on le ressort */ if (cache.data) { return new Response(cache.data, { headers: corsHeaders }); } return new Response( JSON.stringify({ status: 'error', message: 'API error', code: res.status }), { headers: corsHeaders, status: 502 } ); } } catch (err) { if (cache.data) { return new Response(cache.data, { headers: corsHeaders }); } return new Response( JSON.stringify({ status: 'error', message: err.message }), { headers: corsHeaders, status: 500 } ); } } };
